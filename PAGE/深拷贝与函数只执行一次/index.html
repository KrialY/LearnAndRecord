    // 数组[]和对象{}的深拷贝
// 基础类型直接赋值
function deepClone(obj){
  let newObj=obj instanceof Array?[]:{};

  for(let item in obj){
    let temp=typeof obj[item] === "object"?deepClone(obj[item]):obj[item];
    newObj[item]=temp;
  }
  return newObj;
}

    只执行一次

    // 利用闭包的特性
function once(fn){
  let key=true;

  return function(){
    if(key){
      fn.apply(this,arguments);
      key=false;
    }
  }
}
