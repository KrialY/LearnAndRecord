    let divs = document.getElementsByTagName('div');
    Array.prototype.slice2 = function (start, end) {
        let res = new Array();
        let s = start || 0;
        let e = end || this.length;
        for (let i = s;i < e;i++) {
            res.push(this[i]);
        }
        return res;
    }
    console.log(divs, [].slice2.apply(divs), Array.from(divs));
